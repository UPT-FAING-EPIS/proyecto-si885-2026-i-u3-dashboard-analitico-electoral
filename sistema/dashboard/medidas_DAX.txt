========== PÁGINA 1: RESUMEN ==========

[Total Candidatos]
Total Candidatos = COUNTROWS(Candidato)

[Total Varones]
Total Varones = COUNTROWS(FILTER(Candidato, Candidato[Sexo] = "Masculino"))

[Total Damas]
Total Damas = COUNTROWS(FILTER(Candidato, Candidato[Sexo] = "Femenino"))

[% Varones]
% Varones = DIVIDE([Total Varones], [Total Candidatos], 0)

[% Damas]
% Damas = DIVIDE([Total Damas], [Total Candidatos], 0)

[Candidatos con Riesgo]
Candidatos con Riesgo =
CALCULATE(
    COUNTROWS(ClasificacionLegalPenal),
    ClasificacionLegalPenal[NivelPrincipal] >= 2
)

[% Con Riesgo]
% Con Riesgo = DIVIDE([Candidatos con Riesgo], [Total Candidatos], 0)

[Candidatos con Posgrado]
Candidatos con Posgrado =
CALCULATE(
    DISTINCTCOUNT(Posgrado[IdCandidato]),
    Posgrado[Concluido] = TRUE()
)

[% Con Posgrado]
% Con Posgrado = DIVIDE([Candidatos con Posgrado], [Total Candidatos], 0)

[Candidatos con TikTok]
Candidatos con TikTok =
CALCULATE(
    COUNTROWS(CandidatoRedSocial),
    CandidatoRedSocial[Tiene] = TRUE()
)

========== PÁGINA 2: INGRESOS Y PATRIMONIO ==========

[Ingreso Promedio]
Ingreso Promedio = AVERAGE(Ingresos[Total])

[Ingreso Maximo]
Ingreso Maximo = MAX(Ingresos[Total])

[Ingreso Minimo]
Ingreso Minimo = MIN(Ingresos[Total])

[Patrimonio Total por Candidato]
Patrimonio por Candidato =
SUMX(
    VALUES(Bien[IdCandidato]),
    CALCULATE(SUM(Bien[ValorAutovaluo]))
)

[Patrimonio Promedio]
Patrimonio Promedio =
AVERAGEX(
    VALUES(Bien[IdCandidato]),
    CALCULATE(SUM(Bien[ValorAutovaluo]))
)

[Total Inmuebles]
Total Inmuebles =
CALCULATE(
    COUNTROWS(Bien),
    Bien[TipoBien] = "INMUEBLE"
)

[Total Vehiculos]
Total Vehiculos =
CALCULATE(
    COUNTROWS(Bien),
    Bien[TipoBien] = "MUEBLE"
)

[Valor Total Inmuebles]
Valor Total Inmuebles =
CALCULATE(
    SUM(Bien[ValorAutovaluo]),
    Bien[TipoBien] = "INMUEBLE"
)

[Valor Total Vehiculos]
Valor Total Vehiculos =
CALCULATE(
    SUM(Bien[ValorAutovaluo]),
    Bien[TipoBien] = "MUEBLE"
)

========== PÁGINA 4: RIESGO LEGAL ==========

[Riesgo Nivel Critico]
Riesgo Nivel Critico =
CALCULATE(
    COUNTROWS(ClasificacionLegalPenal),
    ClasificacionLegalPenal[NivelPrincipal] = 4
        || ClasificacionLegalPenal[NivelPrincipal] = 5
)

[Riesgo Nivel Alto]
Riesgo Nivel Alto =
CALCULATE(
    COUNTROWS(ClasificacionLegalPenal),
    ClasificacionLegalPenal[NivelPrincipal] = 3
)

[Riesgo Nivel Medio]

    COUNTROWS(ClasificacionLegalPenal),
    ClasificacionLegalPenal[NivelPrincipal] = 2
)

[Riesgo Nivel Bajo]

    COUNTROWS(ClasificacionLegalPenal),
    ClasificacionLegalPenal[NivelPrincipal] = 1
)

[Sin Riesgo]
Sin Riesgo =
CALCULATE(
    COUNTROWS(ClasificacionLegalPenal),
    ClasificacionLegalPenal[NivelPrincipal] = 0
)

================================================================
COLUMNAS CALCULADAS (agregar en la tabla Candidato)
================================================================

[Tiene Posgrado]
Tiene Posgrado =
IF(
    CALCULATE(COUNTROWS(Posgrado), Posgrado[Concluido] = TRUE()) > 0,
    "Con Posgrado",
    "Sin Posgrado"
)

[Nivel Riesgo Texto]
Nivel Riesgo Texto =
VAR nivel = CALCULATE(MAX(ClasificacionLegalPenal[NivelPrincipal]))
RETURN
SWITCH(
    TRUE(),
    nivel >= 4, "Crítico",
    nivel = 3,  "Alto",
    nivel = 2,  "Medio",
    nivel = 1,  "Bajo",
    nivel = 0,  "Sin alerta",
    "Sin clasificar"
)

[Rango Edad]
Rango Edad =
SWITCH(
    TRUE(),
    Candidato[Edad] < 50,              "Menor de 50",
    Candidato[Edad] >= 50 && Candidato[Edad] <= 59, "50 a 59",
    Candidato[Edad] >= 60 && Candidato[Edad] <= 69, "60 a 69",
    Candidato[Edad] >= 70,             "70 o más",
    "Sin dato"
)

================================================================
NOTAS IMPORTANTES
================================================================
- Las medidas de Edad asumen que tienes una columna [Edad] en Candidato.
  Si no existe, créala como columna calculada:
  Edad = DATEDIFF(Candidato[FechaNacimiento], TODAY(), YEAR)
  O agrégala manualmente desde la fuente SQL.

- Formato recomendado para KPIs de porcentaje: Porcentaje, 1 decimal
- Formato para ingresos/patrimonio: Moneda, sin decimales, símbolo S/
- Formato para conteos: Número entero
